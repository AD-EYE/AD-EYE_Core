6/* Avoid empty translation unit warning (ANSI C). */



7typedef int dummy_t;

9#ifdef HOST_CAPI_BUILD



10#include "mex.h"

12#include "rtw_modelmap.h"



13#include "Emulate_Prescan_2019a_capi_host.h"



14#define ARRAY_SIZE(x)                  ( sizeof(x) / sizeof((x)[0]) )



15#define FOUND                          (1)



16#define NOT_FOUND                      (0)



17



18const char *tuningVariableNamesInRegistry[] = {



19  0                                    /* Declare at least one array element. */



20};



21



22int_T prescan_CAPI_find(rtwCAPI_ModelMappingInfo* capiMap, const char_T



23  * tuningVariable)



24{



25  uint_T nModelParams = rtwCAPI_GetNumModelParameters(capiMap);



26  const rtwCAPI_ModelParameters* modelParams = rtwCAPI_GetModelParameters



27    (capiMap);



28  uint_T idx;



29  const char_T* paramName;



30  for (idx=0; idx < nModelParams; idx++) {



31    paramName = rtwCAPI_GetModelParameterName(modelParams, idx);



32    if (strcmp(paramName,tuningVariable) == 0) {



33      return FOUND;



34    }



35  }



36



37  return NOT_FOUND;



38}



39



40void mexFunction(int numberOfOutputs, mxArray *outputs[], int numberOfInputs,



41                 const mxArray *inputs[])



42{



43  rtwCAPI_ModelMappingInfo* capiMap = NULL;



44  Emulate_Prescan_2019a_host_DataMapInfo_T dataMap;



45  uint_T warningCount = 0;



46  uint_T idx = 0;



47



48  /* Check for proper number of arguments. */



49  if (numberOfInputs != 0) {



50    mexErrMsgIdAndTxt( "Prescan:TestAutomation", "Too many input arguments.");



51  }



52



53  if (numberOfOutputs > 1) {



54    mexErrMsgIdAndTxt( "Prescan:TestAutomation", "Too many expected outputs.");



55  }



56



57  /* Initialize CAPI. */



58  Emulate_Prescan_2019a_host_InitializeDataMapInfo(&dataMap, 0);



59  capiMap = &dataMap.mmi;



60  for (idx = 0; idx < ARRAY_SIZE(tuningVariableNamesInRegistry) - 1; idx++) {



61    if (prescan_CAPI_find(capiMap, tuningVariableNamesInRegistry[idx]) ==



62        NOT_FOUND) {



63      mexPrintf("#### Warning: Loss of tunability detected for variable: <%s>.\n",



64                tuningVariableNamesInRegistry[idx]);



65      warningCount++;



66    }



67  }



68



69  outputs[0] = mxCreateDoubleScalar(warningCount);



70}
