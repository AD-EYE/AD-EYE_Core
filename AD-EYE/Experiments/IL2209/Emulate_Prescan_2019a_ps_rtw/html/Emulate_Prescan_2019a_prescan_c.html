11void* prescan_CAPI_get_address_by_name(rtwCAPI_ModelMappingInfo* capiMap, const



12  char_T* tuningVariable)



13{



14  uint_T nModelParams = rtwCAPI_GetNumModelParameters(capiMap);



15  const rtwCAPI_ModelParameters* modelParams = rtwCAPI_GetModelParameters



16    (capiMap);



17  uint_T idx;



18  uint_T addrIdx;



19  const char_T* paramName;



20  for (idx=0; idx < nModelParams; idx++) {



21    paramName = rtwCAPI_GetModelParameterName(modelParams, idx);



22    if (strcmp(paramName, tuningVariable) == 0) {



23      void** dataAddrMap = rtwCAPI_GetDataAddressMap(capiMap);



24      addrIdx = rtwCAPI_GetModelParameterAddrIdx(modelParams, idx);



25      return (void *) rtwCAPI_GetDataAddress(dataAddrMap, addrIdx);



26    }



27  }



28



29  return 0;



30}

32void Emulate_Prescan_2019a_prescan_parameters(RT_MODEL_Emulate_Prescan_2019_T *



33  const Emulate_Prescan_2019a_M)



34{



35  rtwCAPI_ModelMappingInfo* capiMap = &(rtmGetDataMapInfo



36    (Emulate_Prescan_2019a_M).mmi);



37  char csName[256], experimentFileName[256];



38  size_t csNameLen;



39  prescan_object_t* experimentHandle;



40  prescan_object_t* tunableDataHandle;



41  const char pbExt[] = ".pb";



42  prescan_bool_t isRegenerationNeeded;



43  prescan_bool_t isApiSuccessful;



44  strcpy(csName,"Emulate_Prescan_2019a");



45  csNameLen = strlen(csName);



46  csName[csNameLen-3]= '\0';



47  strcpy(experimentFileName, strcat(csName,pbExt));



48  if (PRESCAN_SUCCESS != prescan_experiment_create_from_file(experimentFileName,



49       &experimentHandle)) {



50    printf("ERROR: %s\nAborting simulation.\n",



51           prescan_error_get_last_error_message());



52    exit(1);



53  }



54



55  /* Check consistency of tunable data */



56  if (PRESCAN_SUCCESS != prescan_tunabledata_create(experimentHandle,



57       &tunableDataHandle)) {



58    printf("ERROR: %s\nAborting simulation.\n",



59           prescan_error_get_last_error_message());



60    prescan_experiment_destroy(&experimentHandle);



61    exit(1);



62  }



63



64  isApiSuccessful = prescan_tunabledata_is_regeneration_needed(tunableDataHandle,



65    &isRegenerationNeeded);



66  if (PRESCAN_SUCCESS != isApiSuccessful || isRegenerationNeeded) {



67    printf("ERROR: %s\nAborting simulation.\n",



68           prescan_error_get_last_error_message());



69    prescan_tunabledata_destroy(&tunableDataHandle);



70    prescan_experiment_destroy(&experimentHandle);



71    exit(1);



72  }



73



74  prescan_tunabledata_destroy(&tunableDataHandle);



75



76  /* Reset errors before adding tunable data */



77  prescan_error_last_error_reset();



78  prescan_experiment_destroy(&experimentHandle);



79



80  {



81    const char* errorMessage = prescan_error_get_last_error_message();



82    if (errorMessage) {



83      printf("%s\n", errorMessage);



84      prescan_error_last_error_reset();



85    }



86  }



87}
